多选-调用

            var data =  [{id: 'value0',text: 'Text to display'},{id:"value1",text:"test"}];
            $("select").select2({
                    placeholder: "请选择",
                    tags: true,                 //回车可增加标签
                    multiple: true,
                    maximumSelectionLength: 3,  //最多可选择数量
                    data:data,
                    theme: "classic",           //皮肤
            });
